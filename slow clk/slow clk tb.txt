entity Slow_Clk_Sim is

architecture Behavioral of Slow_Clk_Sim is

signal Clk_in,Clk_out : STD_LOGIC := '0';
component Slow_Clk
Port ( Clk_in : in STD_LOGIC;
       Clk_out : out STD_LOGIC);
end component;
begin
UUT : Slow_Clk 
    PORT MAP(
        Clk_in => Clk_in,
        Clk_out => Clk_out
    );

process begin
    Clk_in <= '0'; wait for 2 ns;
    Clk_in <= '1'; wait for 2 ns;
end process;
end Behavioral;
